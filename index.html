password_gen.py

Gelişmiş Şifre Yöneticisi

Güvenlik analizi yapan ve karmaşık karakter setleri kullanan script.

Kodu İncele 
import random
import string

def create_secure_pw(length=16, use_special=True):
    pool = string.ascii_letters + string.digits
    if use_special:
        pool += "!@#$%^&*()_+-="
    
    password = "".join(random.choice(pool) for _ in range(length))
    
    # Güvenlik Kontrolü
    has_upper = any(c.isupper() for c in password)
    has_lower = any(c.islower() for c in password)
    has_digit = any(c.isdigit() for c in password)
    
    score = sum([has_upper, has_lower, has_digit])
    status = "Güçlü" if score == 3 and length >= 12 else "Zayıf"
    
    return f"Şifre: {password} | Durum: {status}"

print(create_secure_pw(20))

Hava Durumu Çekici

Dış kaynaklı API kullanarak anlık hava durumu verisi işleme.

Kodu İncele
import requests

def get_weather(city):
    api_key = "YOUR_API_KEY"
    url = f"http://api.openweathermap.org/data/2.5/weather?q={city}&appid={api_key}&units=metric"
    
    response = requests.get(url)
    if response.status_code == 200:
        data = response.json()
        temp = data['main']['temp']
        desc = data['weather'][0]['description']
        return f"{city} için Sıcaklık: {temp}°C | Durum: {desc}"
    return "Şehir bulunamadı."

print(get_weather("Ankara"))

Otomatik Dosya Düzenleyici

Kodu İncele
import os
import shutil

def organize_files(path):
    for filename in os.listdir(path):
        name, ext = os.path.splitext(filename)
        ext = ext[1:].lower()
        if ext == "": continue
        
        if not os.path.exists(path + "/" + ext):
            os.makedirs(path + "/" + ext)
        shutil.move(path+"/"+filename, path+"/"+ext+"/"+filename)

Kodu İncele
public class Account {
    public decimal Balance { get; set; }
    public void Deposit(decimal amount) {
        if(amount > 0) Balance += amount;
    }
    public bool Withdraw(decimal amount) {
        if(amount <= Balance) {
            Balance -= amount;
            return true;
        }
        return false;
    }
}
